              .animate({left: "+=400"}, function() {
                  this.animate({rotate:   "+=180", top   :  "+=50"}, function() {
                      this.animate({left: "-=400"                 }, {id      : "back"});
                      this.animate({top :  "-=50", rotate: "+=180"}, {appendTo: "back"});
                  });
              });
